Que- "Write a callback function to print numbers from 1 to 7, in which 1 should be printed after 1 sec, 2 should be printed after 2 sec, 3 should be printed after 3 sec, and so on. Explain callback hell. Numbers 1 2 3 4 5 6 7"

ans- function printNumbersWithDelayUsingCallbacks() {
  let num = 1;

  function printNumberWithDelay(delay, callback) {
    setTimeout(function () {
      console.log(num);
      num++;
      callback();
    }, delay);
  }

  printNumberWithDelay(1000, function () {
    printNumberWithDelay(2000, function () {
      printNumberWithDelay(3000, function () {
        printNumberWithDelay(4000, function () {
          printNumberWithDelay(5000, function () {
            printNumberWithDelay(6000, function () {
              printNumberWithDelay(7000, function () {
                // No more numbers to print
              });
            });
          });
        });
      });
    });
  });
}

printNumbersWithDelayUsingCallbacks();

Que- "Create a promise function accepting an argument, if yes is passed to the function then it should go to resolved state and print Promise Resolved, and if nothing is passed 
then it should go to reject the state and catch the error and print Promise Rejected "


ans- function check(arg) {
  return new Promise((resolve, reject) => {
    if (arg === 'yes') {
      resolve('Promise Resolved');
    } else {
      reject(new Error('Promise Rejected'));
    }
  });
}


check('yes')
  .then((result) => {
    console.log(result);
  })
  .catch((error) => {
    console.error(error.message);
  });

check()
  .then((result) => {
    console.log(result); 
  })
  .catch((error) => {
    console.error(error.message); 
  });

Que- Create examples to explain promise.all function?


ans- 
function delay(rukobhai) {
  return new Promise(resolve => setTimeout(resolve, rukobhai));
}


const task1 = delay(1000).then(() => 'Task 1');
const task2 = delay(2000).then(() => 'Task 2');
const task3 = delay(1500).then(() => 'Task 3');

Promise.all([task1, task2, task3])
  .then(results => {
    console.log('All tasks completed:', results);
  })
  .catch(error => {
    console.error('Error:', error);
  });
